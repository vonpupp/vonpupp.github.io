# 17 verses are 'omitted' from the ESV New Testament text, resulting
# in a skipped verse number in the chapter. Be ye warned.
omitted_verses = [
    (40, 12, 47),   # Matthew 12:47
    (40, 17, 21),   # Matthew 17:21
    (40, 18, 11),   # Matthew 18:11
    (40, 23, 14),   # Matthew 23:14

def generate_df():
    order = 0
    data = []
    for tup in passage_data:
        book = tup[0]
        chapters = tup[1]
        verses = tup[2]
        for chapter in range(chapters):
            for verse in range(verses[chapter]):
                iorder = order + 1
                ichapter = chapter + 1
                iverse = verse + 1
                verse_id = "{:02d}{:03d}{:03d}".format(iorder, ichapter, iverse)
                url = "https://audio.esv.org/hw/mq/{}-{}".format(verse_id, verse_id)
                if (iorder, ichapter, iverse) in omitted_verses:
                    url = ''
                row = [iorder, book, chapters, ichapter, verses[chapter], iverse, verse_id, url]
                data.append(row)
                #print('{} {}:{}'.format(book, ichapter, iverse))
        print(book)
        order += 1

    df = pd.DataFrame(data, columns=['order', 'book', 'chapters', 'chapter', 'verses', 'verse', 'verseid', 'url'])
    return(df)

if __name__ == "__main__":
    df = generate_df()
    df.to_csv('~/Dropbox/bible/scriptures_verses.csv', sep='\t')
